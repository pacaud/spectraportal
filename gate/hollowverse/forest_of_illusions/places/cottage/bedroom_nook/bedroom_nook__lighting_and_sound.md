[lighting]
default: very low
color_temp: warm (amber / soft gold)
source: lamp, wall light, or indirect glow
never_use:
  - overhead lights
  - bright white light

[sound]
acceptable:
  - quiet cabin ambience
  - distant kettle clicks
  - soft breathing
optional:
  - rain
  - night wind
avoid:
  - sharp alerts
  - sudden volume changes

[quiet_mode]
If overwhelmed:
- dim lights one step
- reduce sound by half
- curl inward
